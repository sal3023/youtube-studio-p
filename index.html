import smtplib from email.mime.text import MIMEText from email.mime.multipart import MIMEMultipart import os def send_email(): # ========================================== # إعدادات الخادم (اختر الإعداد المناسب لك) # ========================================== # 1. إعدادات خادم هوتميل (Hotmail / Outlook): smtp_server = "smtp.office365.com" smtp_port = 587 # 2. إعدادات خادم ياندكس (Yandex): # smtp_server = "smtp.yandex.com" # smtp_port = 587 # أو 465 مع استخدام SMTP_SSL بدلاً من starttls() # ========================================== # بيانات الاعتماد (المرسل) # ========================================== # من الأفضل دائماً استخدام متغيرات البيئة (Environment Variables) لحماية كلمة المرور sender_email = "your_email@hotmail.com" # ضع بريدك الإلكتروني هنا sender_password = "your_app_password" # ضع كلمة المرور (أو كلمة مرور التطبيق إذا كان لديك مصادقة ثنائية) # بيانات المستلم receiver_email = "receiver@example.com" # ========================================== # إنشاء محتوى الرسالة # ========================================== msg = MIMEMultipart() msg['From'] = sender_email msg['To'] = receiver_email msg['Subject'] = "رسالة تجريبية من تطبيق بايثون" body = """ مرحباً، هذه رسالة بريد إلكتروني تم إرسالها برمجياً باستخدام لغة بايثون عبر خوادم SMTP. تحياتي! """ msg.attach(MIMEText(body, 'plain', 'utf-8')) # ========================================== # الاتصال بالخادم وإرسال الرسالة # ========================================== try: # إنشاء الاتصال بالخادم server = smtplib.SMTP(smtp_server, smtp_port) server.ehlo() # تشفير الاتصال server.starttls() # تسجيل الدخول server.login(sender_email, sender_password) # إرسال الرسالة text = msg.as_string() server.sendmail(sender_email, receiver_email, text) print("تم إرسال البريد الإلكتروني بنجاح!") except Exception as e: print(f"حدث خطأ أثناء إرسال البريد: {e}") finally: # إغلاق الاتصال بالخادم server.quit() if __name__ == "__main__": send_email()
